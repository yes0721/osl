#include <stdio.h>
#define MAX_FRAMES 10
#define MAX_PAGES 100

int main() {
    int frames, numPages;
    int pages[MAX_PAGES];
    int frame[MAX_FRAMES];
    int counter[MAX_FRAMES]; // To track how recently each frame was used
    int i, j, k, pageFaults = 0;

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    printf("Enter number of pages: ");
    scanf("%d", &numPages);

    printf("Enter the page reference string (space-separated):\n");
    for (i = 0; i < numPages; i++) {
        scanf("%d", &pages[i]);
    }

    // Initialize all frames and counters
    for (i = 0; i < frames; i++) {
        frame[i] = -1;
        counter[i] = 0;
    }

    printf("\nPage\tFrames\t\tPage Fault\n");
    printf("---------------------------------------\n");

    for (i = 0; i < numPages; i++) {
        int page = pages[i];
        int found = 0;

        // Check if page is already in frame
        for (j = 0; j < frames; j++) {
            if (frame[j] == page) {
                found = 1;
                counter[j] = 0; // Reset usage (most recently used)
            } else if (frame[j] != -1) {
                counter[j]++; // Increase usage for other frames
            }
        }

        // If page not found â†’ page fault
        if (!found) {
            int maxCounter = -1;
            int replaceIndex = -1;

            // Find empty frame or least recently used frame
            for (j = 0; j < frames; j++) {
                if (frame[j] == -1) { // Empty slot
                    replaceIndex = j;
                    break;
                }
                if (counter[j] > maxCounter) {
                    maxCounter = counter[j];
                    replaceIndex = j;
                }
            }

            frame[replaceIndex] = page;
            counter[replaceIndex] = 0;
            pageFaults++;

            // Increase counter for others (except replaced one)
            for (j = 0; j < frames; j++) {
                if (j != replaceIndex && frame[j] != -1)
                    counter[j]++;
            }
        }

        // Print frame status
        printf("%d\t", page);
        for (k = 0; k < frames; k++) {
            if (frame[k] == -1)
                printf("- ");
            else
                printf("%d ", frame[k]);
        }
        printf("\t%s\n", found ? "No" : "Yes");
    }

    printf("\nTotal Page Faults = %d\n", pageFaults);
    return 0;
}
