#include <stdio.h>

int main() {
    int frames, pages[100], frame[10];
    int numPages, i, j, k, pageFaults = 0;
    int oldest = 0; // Points to the oldest frame (to be replaced)
    int found;

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    printf("Enter number of pages: ");
    scanf("%d", &numPages);

    printf("Enter the page reference string (space-separated):\n");
    for (i = 0; i < numPages; i++) {
        scanf("%d", &pages[i]);
    }

    // Initialize frame array to -1 (indicates empty)
    for (i = 0; i < frames; i++) {
        frame[i] = -1;
    }

    printf("\nPage\tFrames\t\tPage Fault\n");
    printf("--------------------------------------\n");

    for (i = 0; i < numPages; i++) {
        found = 0;

        // Check if page is already in frame
        for (j = 0; j < frames; j++) {
            if (frame[j] == pages[i]) {
                found = 1;
                break;
            }
        }

        // If not found, replace the oldest page
        if (!found) {
            frame[oldest] = pages[i];
            oldest = (oldest + 1) % frames; // Circular replacement
            pageFaults++;
        }

        // Print current frame status
        printf("%d\t", pages[i]);
        for (k = 0; k < frames; k++) {
            if (frame[k] == -1)
                printf("- ");
            else
                printf("%d ", frame[k]);
        }
        printf("\t%s\n", found ? "No" : "Yes");
    }

    printf("\nTotal Page Faults = %d\n", pageFaults);
    return 0;
}
